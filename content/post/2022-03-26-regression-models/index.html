## Second method: Bootstrapping

set.seed(1)

boot.fn <- function(data, index) {
  return(coef(lm(Price~Cut + Carat, data = diamond_data, subset = index)))
}

boot.out <- boot(diamond_data, boot.fn, 500)
plot(boot.out)
